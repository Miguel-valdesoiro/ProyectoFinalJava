DROP DATABASE IF EXISTS curso_sql;
CREATE DATABASE curso_sql;
USE curso_sql;

DROP TABLE IF EXISTS CLIENTES;
CREATE TABLE IF NOT EXISTS `CLIENTES` (
'id_CodigoCliente'INT NOT NULL, 
'Empresa'VARCHAR(50) NOT NULL,
'Direccion'VARCHAR(50) NOT NULL,
'Telefono'INT NULL,
'Responsable'VARCHAR(50) NOT NULL,
'Hitorial'VARCHAR(20) NOT NULL,
 PRIMARY KEY (`id_CodigoCliente`))
ENGINE = InnoDB;

DROP TABLE IF EXISTS PRODUCTOS;
CREATE TABLE IF NOT EXISTS `PRODUCTOS` (

'id_CodigoArticulo'INT NOT NULL, 
'Seccion'VARCHAR(50) NOT NULL,
'NombreArticulo'VARCHAR(50) NOT NULL,
'Precio'DATE NOT NULL,
'Fecha'DATE NOT NULL,
'Importado'VARCHAR(50)NOT NULL,
'PaisDeOriger'VARCHAR(50)NOT NULL,
'Foto'VARCHAR(20)NOT NULL,
PRIMARY KEY (`id_CodigoArticulo`))
ENGINE = InnoDB;
